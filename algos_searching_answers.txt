let dictionary = ['Albatross', "Bug", "Car", "Dog"];

const linearSearch = (arr, word) => {
    let result = null, value;
    for (let i = 0; i < dictionary.length; i++) {
        value = dictionary[i];
        if(value === word) {
          result = i;
        }

    }
  return result;
}

console.log(linearSearch(['Albatross', "Bug", "Car", "Dog"], "Albatross"));
